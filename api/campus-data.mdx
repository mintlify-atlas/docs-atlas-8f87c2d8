## Overview

The Campus Data API provides access to university-specific information including degree programs, campus locations, navigation routes, and services available to students.

## Degree Programs

Retrieve the list of available degree programs at CUCEI.

### Endpoint

```http
GET /CUB_degrees?order=var_name&dir=ASC
```

### Parameters

<ParamField query="order" type="string">
  Field to sort by (default: `var_name`)
</ParamField>

<ParamField query="dir" type="string">
  Sort direction: `ASC` or `DESC` (default: `ASC`)
</ParamField>

### Response

Returns an array of degree objects:

<ResponseField name="var_code" type="string">
  Unique degree program code (primary key)
</ResponseField>

<ResponseField name="var_name" type="string">
  Full name of the degree program
</ResponseField>

### Example

<CodeGroup>

```typescript TypeScript
import { listDegrees } from './api';

const degrees = await listDegrees();

degrees.forEach(degree => {
  console.log(`${degree.var_code}: ${degree.var_name}`);
});
```

```typescript UI Helper

// Returns: [{ name: "Ingeniería en Computación", code: "INCO" }, ...]

});
```

```typescript TypeScript
export const getJsons = (url: string): Promise<any> => {
  return fetch(url)
    .then(response => response.json())
    .then(json => json)
    .catch(error => {
      console.error('Error fetching JSON:', error);
      throw error;
    });
};
```

## Integration Examples

### Registration Form with Degrees

<CodeGroup>

```typescript TypeScript
import { get_degrees } from './api/consultas';
import { registerUser } from './api/register';

async function registerWithDegree(
  email: string,
  username: string,
  password: string,
  name: string,
  lastnames: string,
  degreeCode: string
) {
  // Fetch available degrees for validation
  const degrees = await get_degrees();
  const validDegree = degrees.find(d => d.code === degreeCode);
  
  if (!validDegree) {
    throw new Error('Invalid degree code');
  }
  
  // Register user with validated degree
  const result = await registerUser({
    email,
    username,
    password,
    name,
    lastnames,
    degree_code: degreeCode
  });
  
  return result;
}
```

```jsx React Component
import { useState, useEffect } from 'react';
import { get_degrees } from './api/consultas';

function RegistrationForm() {
  const [degrees, setDegrees] = useState([]);
  const [formData, setFormData] = useState({
    email: '',
    username: '',
    password: '',
    name: '',
    lastnames: '',
    degree_code: ''
  });
  
  useEffect(() => {
    // Load degrees on mount
    get_degrees().then(setDegrees);
  }, []);
  
  const handleSubmit = async (e) => {
    e.preventDefault();
    // Submit registration with selected degree
    const result = await registerUser(formData);
    
    if (result.success) {
      alert('Registration successful!');
    }
  };
  
  return (
    <form onSubmit={handleSubmit}>
      {/* Other form fields... */}
      
      <select 
        value={formData.degree_code}
        onChange={(e) => setFormData({
          ...formData,
          degree_code: e.target.value
        })}
      >
        <option value="">Select your degree</option>
        {degrees.map(d => (
          <option key={d.code} value={d.code}>
            {d.name}
          </option>
        ))}
      </select>
      
      <button type="submit">Register</button>
    </form>
  );
}
```

### Display User's Degree

<CodeGroup>

```typescript TypeScript
import { findUserById } from './api';
import { listDegrees } from './api';

async function getUserWithDegree(userId: number) {
  const user = await findUserById(userId);
  const degrees = await listDegrees();
  
  const degree = degrees.find(d => d.var_code === user.var_degree_code);
  
  return {
    ...user,
    degree: degree ? degree.var_name : 'Unknown'
  };
}

// Usage
const userData = await getUserWithDegree(42);
console.log(`${userData.var_name} estudia ${userData.degree}`);
// Output: "Juan estudia Ingeniería en Computación"
```

## Caching Strategy

Degree programs rarely change, so implement caching:

<CodeGroup>

```typescript TypeScript
let cachedDegrees: DegreeMapped[] | null = null;
let cacheTimestamp: number = 0;
const CACHE_DURATION = 24 * 60 * 60 * 1000; // 24 hours

async function getDegreesWithCache(): Promise<DegreeMapped[]> {
  const now = Date.now();
  
  // Return cached data if still valid
  if (cachedDegrees && (now - cacheTimestamp) < CACHE_DURATION) {
    console.log('Returning cached degrees');
    return cachedDegrees;
  }
  
  // Fetch fresh data
  console.log('Fetching fresh degree data');
  const degrees = await get_degrees();
  
  // Update cache
  cachedDegrees = degrees;
  cacheTimestamp = now;
  
  return degrees;
}
```

```typescript AsyncStorage (React Native)
import AsyncStorage from '@react-native-async-storage/async-storage';
import { get_degrees } from './api/consultas';

const DEGREES_CACHE_KEY = 'cached_degrees';
const CACHE_DURATION = 24 * 60 * 60 * 1000;

async function getDegreesWithStorage() {
  try {
    // Check cache
    const cached = await AsyncStorage.getItem(DEGREES_CACHE_KEY);
    
    if (cached) {
      const { data, timestamp } = JSON.parse(cached);
      
      if (Date.now() - timestamp < CACHE_DURATION) {
        return data;
      }
    }
    
    // Fetch fresh data
    const degrees = await get_degrees();
    
    // Store in cache
    await AsyncStorage.setItem(DEGREES_CACHE_KEY, JSON.stringify({
      data: degrees,
      timestamp: Date.now()
    }));
    
    return degrees;
  } catch (error) {
    console.error('Cache error:', error);
    // Fallback to direct fetch
    return get_degrees();
  }
}
```

</CodeGroup>

## Best Practices

<AccordionGroup>
  <Accordion title="Cache degree data locally">
    Degree programs don't change frequently, so cache them:
    - Reduces API calls
    - Improves app performance
    - Works offline
    
    Refresh cache every 24 hours or on app update.
  </Accordion>
  
  <Accordion title="Validate degree codes before submission">
    Always validate user-selected degree codes:
    ```typescript
    const degrees = await get_degrees();
    const isValid = degrees.some(d => d.code === selectedCode);
    
    if (!isValid) {
      alert('Please select a valid degree program');
      return;
    }
    ```
  </Accordion>
  
  <Accordion title="Provide user-friendly degree names">
    Display full degree names in UI, use codes for API:
    ```typescript
    // Good: Show full name
    <option value="INCO">Ingeniería en Computación</option>
    
    // Avoid: Showing only code
    <option value="INCO">INCO</option>
    ```
  </Accordion>
  
  <Accordion title="Handle missing or invalid degrees gracefully">
    If a user has an invalid degree code:
    ```typescript
    const degreeName = degrees.find(d => d.code === user.var_degree_code)?.name 
      || 'Degree not found';
    ```
  </Accordion>
</AccordionGroup>
