## Authentication System

### Login Flow

The login system validates credentials against Supabase and establishes a session:

**Location**: `Src/auth/LoginScreen.js:196`

```javascript Login Handler
const handleLogin = async () => {
  console.log('🔍 === INICIANDO PROCESO DE LOGIN ===');
  setShowError(false);
  setErrorMessage("");

  // Validate empty fields
  if (!formData.username || !formData.password) {
    setErrorMessage("Por favor, completa todos los campos.");
    setShowError(true);
    return;
  }

  // Validate email format
  if (!validateEmail(formData.username)) {
    setErrorMessage("Ingresa un correo electrónico válido.");
    setShowError(true);
    return;
  }

  setIsLoading(true);
  
  try {
    // Call Supabase login function
    const result = await login(formData.username.trim(), formData.password);
    
    if (result && result.isMatch === true && result.userData) {
      console.log("✅ === LOGIN SÚPER EXITOSO ===");
      console.log("👤 Usuario logueado:", result.userData.username);
      
      // Save session
      await setSession(result.userData);
      
      // Show success animation
      setShowSuccessAnimation(true);
      setModalVisible(true);
      
      setTimeout(() => {
        setModalVisible(false);
        setShowSuccessAnimation(false);
        
        // Navigate to home
        navigation.reset({
          index: 0,
          routes: [
            { name: "Principal Home", params: { user: result.userData } },
          ],
        });
      }, 3000);
    } else {
      setErrorMessage("Correo o contraseña incorrectos.");
      setShowError(true);
    }
  } catch (error) {
    console.error("🚨 === ERROR EN HANDLELOGIN ===", error);
    setErrorMessage("Error al iniciar sesión. Intenta de nuevo.");
    setShowError(true);
  } finally {
    setIsLoading(false);
  }
};
```

## Session Management

### Session Storage

Sessions persist across app restarts using AsyncStorage:

**Location**: `Src/auth/SessionManager.js:4`

```javascript Session Manager
export const setSession = async (userData) => { 
  try {
    console.log('💾 === GUARDANDO SESIÓN ===');
    
    // Validate required fields
    if (!userData || !userData.email || !userData.username) {
      throw new Error('Datos de usuario incompletos para guardar sesión');
    }
    
    const sessionData = JSON.stringify(userData);
    await AsyncStorage.setItem("userSession", sessionData);
    
    // Immediate verification
    const savedData = await AsyncStorage.getItem("userSession");
    if (savedData) {
      const parsedData = JSON.parse(savedData);
      console.log('✅ === VERIFICACIÓN EXITOSA ===');
      console.log('👤 Usuario verificado:', parsedData.username);
      console.log('📧 Email verificado:', parsedData.email);
      return true;
    }
  } catch (error) {
    console.error("🚨 Error guardando sesión:", error);
    throw error;
  }
};

export const getSession = async () => {
  try {
    const session = await AsyncStorage.getItem("userSession");
    
    if (session) {
      const userData = JSON.parse(session);
      console.log('✅ === SESIÓN ENCONTRADA ===');
      console.log('👤 Usuario:', userData.username);
      return userData;
    } else {
      console.log('❌ No hay sesión guardada');
      return null;
    }
  } catch (error) {
    console.error("🚨 Error obteniendo sesión:", error);
    return null;
  }
};

export const clearSession = async () => {
  try {
    await AsyncStorage.removeItem("userSession");
    console.log("🗑️ Sesión eliminada exitosamente");
  } catch (error) {
    console.error("🚨 Error eliminando sesión:", error);
  }
};
```

const getShadowStyle = (elevation) => {
