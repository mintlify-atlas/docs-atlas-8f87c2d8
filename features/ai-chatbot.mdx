### Query Function

**Location**: `Src/Screens/ChatBot/Chatbot.js:143`

```javascript API Request
async function query(user_input) {
  const payload = {
    user_id: "1234",
    "in-0": user_input,
  };

  try {
    const response = await fetch(API_URL, {
      method: "POST",
      headers: HEADERS,
      body: JSON.stringify(payload),
    });

    if (!response.ok) {
      throw new Error("Error al comunicarse con la API");
    }

    const responseData = await response.json();
    const rawResponse =
      responseData.outputs?.["out-0"] ||
      "No se recibió una respuesta válida en el campo 'out-0'.";

    // Limit response to 5000 characters
    return rawResponse.slice(0, 5000);
  } catch (e) {
    return `Error al comunicarse con la API: ${e.message}`;
  }
}
```

```javascript Response Cleaning
function processResponse(response) {
  // Remove citation markers
  response = response.replace(/\[\^.*?\]/g, "");
  // Remove citation blocks
  response = response.replace(/<citations>.*?<\/citations>/gs, "");
  // Collapse multiple newlines
  response = response.replace(/\n\s*\n/g, "\n");
  return response.trim();
}
```

```javascript Greeting Detection
const isGreeting = (message) => {
  const greetings = [
    "hola",
    "hello",
    "hi",
    "hey",
    "buenos días",
    "buenas tardes",
    "buenas noches",
  ];
  return greetings.some((greeting) => 
    message.toLowerCase().includes(greeting)
  );
};

const getGreetingResponse = () => {
  const responses = [

    "¡Hola! ¿En qué puedo asistirte hoy?",
  ];
  return responses[Math.floor(Math.random() * responses.length)];
};
```

## Message Sending Flow

**Location**: `Src/Screens/ChatBot/Chatbot.js:304`

```javascript Send Message Logic
const onSend = useCallback(async () => {
  if (inputMessage.trim() === "") return;

  // Create user message
  const newMessage = {
    _id: `${Date.now()}-${Math.random()}`,
    text: inputMessage,
    createdAt: new Date().toISOString(),
    user: { _id: 1, name: "Usuario" },
  };

  setMessages((prevMessages) => [newMessage, ...prevMessages]);
  setInputMessage("");
  setIsTyping(true);

  try {
    if (isGreeting(inputMessage)) {
      // Fast local response for greetings
      addBotMessage(getGreetingResponse());
    } else {
      // Query Stack AI for other messages
      const responseFromAPI = await query(inputMessage);
      const processedResponse = processResponse(responseFromAPI);
      addBotMessage(processedResponse);
    }
  } catch (error) {
    addBotMessage(
      "Lo siento, ha ocurrido un error. Por favor, intenta de nuevo más tarde."
    );
  } finally {
    setIsTyping(false);
  }
}, [inputMessage]);
```

## Link Detection

Automatic URL detection and rendering:

**Location**: `Src/Screens/ChatBot/Chatbot.js:436`

```javascript Link Rendering
const renderTextWithLinks = (text) => {
  const urlRegex = /(https?:\/\/[^\s]+)/g;
  const parts = text.split(urlRegex);
  
  return parts.map((part, index) => {
    if (part.match(urlRegex)) {
      return (
        <Text
          key={index}
          style={styles.link}
          onPress={() => Linking.openURL(part)}
        >
          {part}
        </Text>
      );
    }
    return <Text key={index}>{part}</Text>;
  });
};
```

## Inactivity Management

Automatic conversation cleanup after inactivity:

**Location**: `Src/Screens/ChatBot/Chatbot.js:476`

```javascript Inactivity Timer
const resetInactivityTimer = () => {
  clearTimeout(inactivityTimerRef.current);
  clearTimeout(deleteTimerRef.current);
  
  // Show warning after 2 minutes
  inactivityTimerRef.current = setTimeout(() => {
    setShowInactivityModal(true);
  }, 2 * 60 * 1000);
  
  // Auto-delete after 3 minutes
  deleteTimerRef.current = setTimeout(() => {
    deleteConversation();
  }, 3 * 60 * 1000);
};

const deleteConversation = async () => {
  const welcomeMessage = messages.find(
    (message) =>
      message.user._id === 2 &&
      message.text.includes("¡Hola! Soy el bot asistente de CUCEI")
  );
  const newMessages = welcomeMessage ? [welcomeMessage] : [];
  setMessages(newMessages);
  await AsyncStorage.setItem("chatMessages", JSON.stringify(newMessages));
  setShowInactivityModal(false);
};
```

## Chatbot Button

Floating button with tooltip and auto-reappear:

**Location**: `Src/Screens/ChatBot/Chatboot_Button.js:19`

```javascript ChatbotButton Component
export const ChatbotButton = () => {
  const navigation = useNavigation();
  const [showTooltip, setShowTooltip] = useState(true);
  const [showBot, setShowBot] = useState(true);
  const appState = useRef(AppState.currentState);
  const timer = useRef(null);

  const REAPPEAR_DELAY = 5 * 60 * 1000; // 5 minutes

  const hideBot = async () => {
    setShowBot(false);
    try {
      await AsyncStorage.setItem('@botVisibility', JSON.stringify(false));
      
      // Auto-reappear after delay
      timer.current = setTimeout(() => {
        resetBotVisibility();
      }, REAPPEAR_DELAY);
    } catch (error) {
      console.error('Error saving bot visibility:', error);
    }
  };

  return (
    <View style={styles.container}>
      {showTooltip && (
        <View style={styles.tooltip}>
          <Text style={styles.tooltipText} onPress={handlePress}>
            ¡Hola! ¿Necesitas ayuda?
          </Text>
          <TouchableOpacity 
            style={styles.closeTooltipButton} 
            onPress={dismissTooltip}
          >
            <FontAwesomeIcon icon={faTimes} size={16} color="#FFFFFF" />
          </TouchableOpacity>
        </View>
      )}
      <View style={styles.buttonContainer}>
        <TouchableOpacity onPress={handlePress}>
          <Image
            source={require('../ChatBot/images/bot.png')}
            style={styles.chatbotIcon}
          />
        </TouchableOpacity>
        <TouchableOpacity 
          style={styles.closeButton} 
          onPress={hideBot}
        >
          <FontAwesomeIcon icon={faTimes} size={16} color="#FFFFFF" />
        </TouchableOpacity>
      </View>
    </View>
  );
};
```
