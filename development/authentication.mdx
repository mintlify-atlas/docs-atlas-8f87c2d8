## Overview

CUCEIUbicate implements a secure authentication system with password hashing, session persistence using AsyncStorage, and comprehensive login/registration flows. The system supports student authentication and maintains user sessions across app restarts.

## Session Management

### SessionManager Module

The `SessionManager` handles all session operations using AsyncStorage for persistent storage.

<CodeGroup>
```javascript Src/auth/SessionManager.js
import AsyncStorage from "@react-native-async-storage/async-storage";

// Save user session
export const setSession = async (userData) => { 
  try {
    console.log('💾 === GUARDANDO SESIÓN ===');
    
    // Validate minimum required data
    if (!userData || !userData.email || !userData.username) {
      throw new Error('Datos de usuario incompletos para guardar sesión');
    }
    
    const sessionData = JSON.stringify(userData);
    await AsyncStorage.setItem("userSession", sessionData);
    
    // Immediate verification
    const savedData = await AsyncStorage.getItem("userSession");
    if (savedData) {
      const parsedData = JSON.parse(savedData);
      console.log('✅ Usuario verificado:', parsedData.username);
      return true;
    }
    return false;
  } catch (error) {
    console.error("🚨 Error guardando sesión:", error);
    throw error;
  }
};

// Retrieve user session
export const getSession = async () => {
  try {
    const session = await AsyncStorage.getItem("userSession");
    
    if (session) {
      const userData = JSON.parse(session);
      console.log('✅ Sesión encontrada:', userData.username);
      return userData;
    }
    return null;
  } catch (error) {
    console.error("🚨 Error obteniendo sesión:", error);
    return null;
  }
};

// Clear user session
export const clearSession = async () => {
  try {
    await AsyncStorage.removeItem("userSession");
    console.log("🗑️ Sesión eliminada exitosamente");
  } catch (error) {
    console.error("🚨 Error eliminando sesión:", error);
  }
};
```
</CodeGroup>

### Login API Integration

The login function communicates with the backend API to authenticate users.

<CodeGroup>
```typescript Src/Api/login.ts
import { findUserByEmail, findUserByUsername, UserRow } from './lib/api';

export const login = async (
  user: string,
  contraseña: string
): Promise<{ isMatch: boolean; userData: any } | false> => {
  try {
    console.log('🔍 === INICIANDO LOGIN ===');
    
    if (!user || !contraseña) {
      return { isMatch: false, userData: null };
    }
    
    // Clean and normalize user input
    const cleanUser = user.normalize('NFD')
      .replace(/[\u0300-\u036f]/g, '')
      .toLowerCase()
      .trim();
    
    // Search by email first
    let rows: UserRow[] = await findUserByEmail(cleanUser);
    
    // If not found, search by username
    if (!rows?.length) {
      rows = await findUserByUsername(cleanUser);
    }
    
    if (!rows?.length) {
      console.log('❌ Usuario no encontrado');
      return { isMatch: false, userData: null };
    }

    const u = rows[0];
    console.log('👤 Usuario encontrado:', u.var_username);
    
    // Compare passwords
    const ok = secureCompare(contraseña, u.var_password);
    
    if (!ok) {
      console.log('❌ Contraseña incorrecta');
      return { isMatch: false, userData: null };
    }

    console.log('✅ LOGIN EXITOSO');

    const userData = {
      id: u.id,
      email: u.var_email,
      username: u.var_username,
      name: u.var_name,
      lastnames: u.var_lastnames,
      degree_code: u.var_degree_code,
      code: u.int_user_code || 'No asignado'
    };
    
    return {
      isMatch: true,
      userData: userData,
    };
  } catch (error) {
    console.error('🚨 ERROR EN LOGIN:', error);
    return false;
  }
};
```
</CodeGroup>

<Note>
The login function uses secure password hashing with a custom `secureCompare` function that validates against stored hashes with format: `$secure$hash1$hash2$hash3$timestamp`
</Note>

const handleRegister = async () => {
  setEmailError(false);

    if (!passwordValidation.isValid) {
      setPasswordError(true);
      setErrorMsg(passwordValidation.message);
      shakeForm();
      throw new Error("Contraseña no válida");
    }

    // Confirm password match
    if (password !== confirmPassword) {
      setPasswordError(true);
      setErrorMsg("Las contraseñas no coinciden");
      shakeForm();
      throw new Error("Las contraseñas no coinciden");
    }

    // Check if email exists
    const correoValido = await validar_correo(email);
    if (!correoValido) {
      setEmailError(true);
      setErrorMsg("Este correo electrónico ya se ha registrado");
      shakeForm();
      throw new Error("Correo ya registrado");
    }

    // Success - navigate to complete profile
    setShowSuccessAnimation(true);
    setModalVisible(true);
    setTimeout(() => {
      setModalVisible(false);
      setShowSuccessAnimation(false);
      navigation.navigate("Completar Perfil", { mail: email, pass: password });

    // Error handled
  } finally {
    setIsLoading(false);
  }
};

## Password Security

### Secure Hash Implementation

CUCEIUbicate uses a custom secure hashing algorithm with salt and timestamp:

```typescript Src/Api/login.ts
const secureHash = (password: string): string => {
  const staticSalt = 'CUCEI_UBICATE_2024_PRODUCTION_SECURE_SALT_V2';
  const timestamp = Date.now().toString(36);
  const combined = password + staticSalt + timestamp.slice(-6);
  
  let hash1 = 0;
  let hash2 = 0;
  let hash3 = 0;
  
  // Generate three separate hashes
  for (let i = 0; i < combined.length; i++) {
    const char = combined.charCodeAt(i);
    hash1 = ((hash1 << 5) - hash1) + char;
    hash1 = hash1 & hash1;
  }
  
  for (let i = 0; i < combined.length; i++) {
    const char = combined.charCodeAt(i);
    hash2 = ((hash2 << 3) - hash2) + char + i;
    hash2 = hash2 & hash2;
  }
  
  const mixed = password + staticSalt;
  for (let i = 0; i < mixed.length; i++) {
    const char = mixed.charCodeAt(i);
    hash3 = ((hash3 << 7) - hash3) + char * (i + 1);
    hash3 = hash3 & hash3;
  }
  
  const finalHash1 = Math.abs(hash1).toString(36).padStart(8, '0');
  const finalHash2 = Math.abs(hash2).toString(36).padStart(8, '0');
  const finalHash3 = Math.abs(hash3).toString(36).padStart(6, '0');
  
  return `$secure$${finalHash1}$${finalHash2}$${finalHash3}$${timestamp.slice(-6)}`;
};
```

<Warning>
The password hashing implementation is custom and specific to this application. For production systems, consider using industry-standard libraries like bcrypt or argon2.
</Warning>

## App Initialization Flow

### Checking Authentication State

The main App component checks authentication state on startup:

<CodeGroup>
```javascript App.js
const checkAppState = async () => {
  try {
    console.log('🔍 Verificando estado de la app...');
    
    // 1. Check onboarding status first
    const onboardingCompleted = await AsyncStorage.getItem("@onboarding_completed");
    
    if (onboardingCompleted === null) {
      console.log('🎯 Primera vez - mostrando onboarding');
      setIsFirstLaunch(true);
      setIsLoggedIn(false);
      return;
    }
    
    // 2. Check session after onboarding
    const session = await getSession();
    
    if (session) {
      console.log('✅ Usuario logueado:', session.username);
      setUserSession(session);
      setIsLoggedIn(true);
      setIsLoginSuccess(true);
      setIsFirstLaunch(false);
    } else {
      console.log('🔐 No hay sesión - ir a login');
      setIsLoggedIn(false);
      setIsFirstLaunch(false);
    }
    
  } catch (error) {
    console.error("🚨 Error verificando estado:", error);
    setIsFirstLaunch(true);
    setIsLoggedIn(false);
  }
};
```

## Logout Implementation

Users can logout from the drawer menu:

```javascript Src/Screens/Home/Components/SearchBarsComponent/MyDrawer.js
const handleLogout = () => {
  Alert.alert(
    "Cerrar Sesión",
    "¿Estás seguro de que quieres cerrar sesión?",
    [
      { text: "Cancelar", style: "cancel" },
      { text: "Sí, cerrar sesión", onPress: confirmLogout, style: "destructive" },

    { cancelable: true }

const confirmLogout = () => {
  clearSession();
  navigation.reset({
    index: 0,
    routes: [{ name: "Login" }],
  });
};
```
