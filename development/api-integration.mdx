### API Client Setup

The base API client handles all HTTP communication:

<CodeGroup>
```typescript Src/Api/api.ts
const BASE = 'https://ilabtdi.com/api_ubicate';
const USE_INDEX_PHP = true;

const url = (p: string) => `${BASE}${USE_INDEX_PHP ? '/index.php' : ''}${p}`;

// String sanitization for collation issues
const cleanString = (str: string): string => {
  if (!str) return '';
  
  return str
    .normalize('NFD')                    // Decompose accented characters
    .replace(/[\u0300-\u036f]/g, '')    // Remove diacritics
    .replace(/[^\w\s@.-]/g, '')         // Only safe characters
    .trim();
};

// Core HTTP function
async function http(path: string, init: RequestInit = {}) {
  try {
    console.log('🌐 Making request to:', url(path));
    
    const res = await fetch(url(path), {
      ...init,
      headers: { 
        'Content-Type': 'application/json', 
        'Accept': 'application/json',
        ...(init.headers || {}) 
      },
    });
    
    console.log('📡 Response status:', res.status);
    
    if (!res.ok) {
      const errorText = await res.text();
      console.error('❌ Server error:', errorText);
      throw new Error(`HTTP ${res.status}: ${errorText}`);
    }
    
    const responseText = await res.text();
    console.log('📄 Raw server response:', responseText);
    
    if (!responseText.trim()) {
      console.log('⚠️ Empty response from server');
      return null;
    }
    
    // Clean response - remove text before JSON
    let cleanedResponse = responseText;
    const jsonStartArray = responseText.indexOf('[');
    const jsonStartObject = responseText.indexOf('{');
    
    let jsonStart = -1;
    if (jsonStartArray !== -1 && jsonStartObject !== -1) {
      jsonStart = Math.min(jsonStartArray, jsonStartObject);
    } else if (jsonStartArray !== -1) {
      jsonStart = jsonStartArray;
    } else if (jsonStartObject !== -1) {
      jsonStart = jsonStartObject;
    }
    
    if (jsonStart > 0) {
      cleanedResponse = responseText.substring(jsonStart);
      console.log('🧹 Cleaned response:', cleanedResponse);
    }
    
    try {
      const jsonData = JSON.parse(cleanedResponse);
      console.log('✅ JSON parsed successfully:', jsonData);
      return jsonData;
    } catch (parseError) {
      console.error('❌ Error parsing JSON:', parseError);
      
      if (cleanedResponse.includes('<html>') || cleanedResponse.includes('<!DOCTYPE')) {
        throw new Error('Server returned HTML instead of JSON. Check URL and endpoint.');
      }
      
      throw new Error(`Invalid server response: ${cleanedResponse.substring(0, 100)}...`);
    }
    
  } catch (error) {
    console.error('🚨 Error in HTTP request:', error);
    throw error;
  }
}
```
</CodeGroup>

<Warning>
The API client includes response cleaning to handle cases where the server returns extra text before JSON. This is common with some PHP configurations.
</Warning>

export const findUserByEmail = async (email: string): Promise<UserRow[]> => {
  try {
    console.log('🔍 Searching user by email:', email);
    
    // Clean and encode email
    const cleanEmail = cleanString(email.toLowerCase());

    if (result.data && Array.isArray(result.data)) return result.data;
    if (typeof result === 'object') return [result];
    
    return [];
  } catch (error) {
    console.error('❌ Error searching user by email:', error);
    return [];
  }
};

### Creating Users

<CodeGroup>
```typescript Insert User
export const insertUser = async (userData: {
  email: string;
  username: string;
  password: string;
  name: string;
  lastnames: string;
  degree_code: string;
}): Promise<number> => {
  try {
    console.log('📝 Inserting new user:', userData.username);
    
    // Hash password
    const hashedPassword = secureHash(userData.password);
    
    // Clean all strings
    const payload = {
      var_email: cleanString(userData.email.toLowerCase()),
      var_password: hashedPassword,
      var_degree_code: cleanString(userData.degree_code.toUpperCase()),
      var_name: cleanString(userData.name),
      var_lastnames: cleanString(userData.lastnames),
      var_username: cleanString(userData.username.toLowerCase())
    };

    const result = await http('/CUB_users', {
      method: 'POST',
      body: JSON.stringify(payload)
    });

    console.log('✅ User inserted, response:', result);

    // Handle different response formats
    if (typeof result === 'number') return result;
    if (result && result.id) return result.id;
    if (result && result.insertId) return result.insertId;
    
    // If success, fetch the newly created user
    if (result && (result.success || result.status === 'success')) {
      const newUser = await findUserByUsername(userData.username);
      if (newUser.length > 0 && newUser[0].id) {
        return newUser[0].id;
      }
    }

    throw new Error('Could not get ID of inserted user');
  } catch (error) {
    console.error('❌ Error inserting user:', error);
    throw error;
  }
};
```

## Validation Utilities

Client-side validation functions:

<Tabs>
  <Tab title="Email">
    ```typescript
    export const validateEmail = (email: string): { isValid: boolean; message?: string } => {
      if (!email) {
        return { isValid: false, message: 'El email es requerido' };
      }

      const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
      if (!emailRegex.test(email)) {
        return { isValid: false, message: 'Formato de email inválido' };
      }

      return { isValid: true };
    };
    ```
  </Tab>
  <Tab title="Username">
    ```typescript
    export const validateUsername = (username: string): { isValid: boolean; message?: string } => {
      if (!username) {
        return { isValid: false, message: 'El nombre de usuario es requerido' };
      }

      if (username.length < 3) {
        return { isValid: false, message: 'Mínimo 3 caracteres' };
      }

      if (username.length > 30) {
        return { isValid: false, message: 'Máximo 30 caracteres' };
      }

      if (!/^[a-zA-Z0-9_.-]+$/.test(username)) {
        return { isValid: false, message: 'Solo letras, números, puntos, guiones y guiones bajos' };
      }

      return { isValid: true };
    };
    ```

## Registration API

### Complete Registration Flow

```typescript Src/Api/register.ts
export const registerUser = async (userData: {
  email: string;
  username: string;
  password: string;
  name: string;
  lastnames: string;
  degree_code: string;
}): Promise<{ success: boolean; message: string; userId?: number }> => {
  try {
    // Validate email
    const emailValidation = validateEmail(userData.email);
    if (!emailValidation.isValid) {
      return { success: false, message: emailValidation.message! };
    }

    // Validate username
    const usernameValidation = validateUsername(userData.username);
    if (!usernameValidation.isValid) {
      return { success: false, message: usernameValidation.message! };
    }

    // Validate password
    const passwordValidation = validatePassword(userData.password);
    if (!passwordValidation.isValid) {
      return { success: false, message: passwordValidation.message! };
    }

    // Check if email exists
    const existingEmail = await findUserByEmail(userData.email);
    if (existingEmail && existingEmail.length > 0) {
      return { success: false, message: 'El email ya está registrado' };
    }

    // Check if username exists
    const existingUsername = await findUserByUsername(userData.username);
    if (existingUsername && existingUsername.length > 0) {
      return { success: false, message: 'El nombre de usuario ya está en uso' };
    }

    // Insert user
    const userId = await insertUser(userData);
    
    return {
      success: true,
      message: 'Usuario registrado exitosamente',
      userId: userId
    };

  } catch (error) {
    console.error('🚨 Error en registro:', error);
    return {
      success: false,
      message: 'Error interno del servidor. Por favor intenta de nuevo.'
    };
  }
};
```
